Polyweight

A color theme without much color.

# Light Theme Example ( python ):

class Polyweight ( ColorTheme ):
    def init ( self, *args, **kwargs ):
        super( ).__init__( )
        self.color = False
        self.themes = [
            'light',
            'dark'
        ]
        self.font = "IBM Plex Mono"
        self.ligatures = "Not yet, but soon"
